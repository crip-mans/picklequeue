⚙️ Setup required in Supabase

Run this SQL once in your Supabase SQL Editor if the clubs table doesn't exist yet:

CREATE TABLE IF NOT EXISTS clubs (
  id         UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id    UUID NOT NULL UNIQUE,
  club_name  TEXT NOT NULL,
  email      TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE clubs ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Public can read clubs"
  ON clubs FOR SELECT USING (true);
CREATE POLICY "Auth users insert own club"
  ON clubs FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Auth users update own club"
  ON clubs FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "Allow anon delete for super admin"
  ON clubs FOR DELETE USING (true);
